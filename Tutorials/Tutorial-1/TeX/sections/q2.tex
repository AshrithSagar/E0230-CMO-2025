\section*{Question 2}

Suppose \( x, y \in \mathbb{R}^{n} \).
Prove the following statements.
Note that \( \Vert \cdot \Vert \) refers to the Euclidean norm if not otherwise specified.
\begin{enumerate}[label=(\alph*), noitemsep]
  \item \( \big\vert \Vert x \Vert - \Vert y \Vert \big\vert \leq \Vert x - y \Vert \leq \Vert x \Vert + \Vert y \Vert \)
  \item \( 2 x^{\top} y = {\Vert x \Vert}^{2} + {\Vert y \Vert}^{2} - {\Vert x - y \Vert}^{2} \)
  \item Suppose \( B \in \mathbb{R}^{m \times n} \).
    Show that the function \( d(B) = \sqrt{\operatorname{Tr}\left( B^{\top} B \right)} \) is a norm (this is the Frobenius norm of a matrix, denoted by \( \Vert \cdot \Vert_{F} \) ).
\end{enumerate}

\subsection*{Solution}

\subsubsection*{(a) \( \big\vert \Vert \mathbf{x} \Vert - \Vert \mathbf{y} \Vert \big\vert \leq \Vert \mathbf{x} - \mathbf{y} \Vert \leq \Vert \mathbf{x} \Vert + \Vert \mathbf{y} \Vert \)}

\subsubsection*{(b) \( 2 \, \mathbf{x}^{\top} \mathbf{y} = {\Vert \mathbf{x} \Vert}^{2} + {\Vert \mathbf{y} \Vert}^{2} - {\Vert \mathbf{x} - \mathbf{y} \Vert}^{2} \)}
