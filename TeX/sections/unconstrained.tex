\chapter{Unconstrained optimisation}

\section{Optimisation methods}

\subsubsection{First-order methods}

Assume \( f \in C^{1} \), and use \( f(\mathbf{x} + \mathbf{p}) = f(\mathbf{x}) + \nabla f(\mathbf{x})^{\top} \mathbf{p} + o(\| \mathbf{p} \|) \).

\subsubsection{Second-order methods}

Assume \( f \in C^{2} \), and use \( f(\mathbf{x} + \mathbf{p}) = f(\mathbf{x}) + \nabla f(\mathbf{x})^{\top} \mathbf{p} + \frac{1}{2} \mathbf{p}^{\top} \nabla^{2} f(\mathbf{x}) \mathbf{p} + o(\| \mathbf{p} \|^2) \).

\subsection{Global minimum}

The point \( \mathbf{x}^* \in \mathbb{R}^d \) is a \textbf{global minimum} of the function \( f: \mathbb{R}^d \to \mathbb{R} \) if
\begin{equation*}
    f(\mathbf{x}^*) \leq f(\mathbf{x}), \quad \forall \mathbf{x} \in \mathbb{R}^d
\end{equation*}

\subsection{Local minimum}

The point \( \mathbf{x}^* \in \mathbb{R}^d \) is a \textbf{local minimum} of the function \( f: \mathbb{R}^d \to \mathbb{R} \) if there exists a \( \delta > 0 \) such that for all \( \mathbf{x} \) in the \( \delta \)-neighborhood of \( \mathbf{x}^* \), we have \( f(\mathbf{x}^*) \leq f(\mathbf{x}) \), i.e.,
\begin{equation*}
    f(\mathbf{x}^*) \leq f(\mathbf{x}), \quad \forall \mathbf{x} \in B_{\delta}(\mathbf{x}^*)
\end{equation*}

\subsection{Necessary and sufficient conditions}

\subsubsection{First-order necessary condition for local minimum}

If \( f \in C^{1} \) and \( \mathbf{x}^* \) is a local minimum of \( f \), then \( \nabla f(\mathbf{x}^*) = \mathbf{0} \).

\subsubsection{Second-order necessary condition for local minimum}

If \( f \in C^{2} \) and \( \mathbf{x}^* \) is a local minimum of \( f \), then \( \nabla f(\mathbf{x}^*) = \mathbf{0}, \; \nabla^2 f(\mathbf{x}^*) \succeq 0 \).

\subsubsection{Second-order sufficient condition for a strict local minimum}

If \( f \in C^{2} \), \( \nabla f(\mathbf{x}^*) = \mathbf{0} \) and \( \nabla^2 f(\mathbf{x}^*) \succ 0 \), then \( \mathbf{x}^* \) is a strict local minimum of \( f \).

\subsubsection{First-order sufficient condition under convexity}

If \( f \in C^1 \) is a convex function and \( \nabla f(\mathbf{x}^*) = \mathbf{0} \), then \( \mathbf{x}^* \) is a global minimum of \( f \).

\subsection{Convex functions}

A function $f: \mathbb{R}^n \to \mathbb{R}$ is \textbf{convex} if
\begin{equation*}
    f(\lambda \mathbf{x} + (1 - \lambda) \mathbf{y})
    \leq
    \lambda f(\mathbf{x}) + (1 - \lambda) f(\mathbf{y}),
    \quad \forall \mathbf{x}, \mathbf{y} \in \mathbb{R}^n, \; \lambda \in [0, 1]
\end{equation*}

\paragraph{First-order condition for convexity:}
If \( f \in C^1 \), then \( f \) is convex iff
\begin{equation*}
    f(\mathbf{y})
    \geq
    f(\mathbf{x}) + \nabla f(\mathbf{x})^\top (\mathbf{y} - \mathbf{x}),
    \quad \forall \mathbf{x}, \mathbf{y} \in \mathbb{R}^n
\end{equation*}

\paragraph{Second-order condition for convexity:}
If \( f \in C^2 \), then \( f \) is convex if
\begin{equation*}
    \nabla^2 f(\mathbf{x}) \succeq 0,
    \quad \forall \mathbf{x} \in \mathbb{R}^n
\end{equation*}

\subsection{Algorithmic design}

\paragraph{Problem:}
\begin{align*}
    \min_{\mathbf{x} \in \mathbb{R}^d} f(\mathbf{x}),
    \quad \text{where }
    f = \left( \frac{1}{2} \mathbf{x}^{\top} \mathbf{Q} \mathbf{x} + \mathbf{h}^{\top} \mathbf{x} + c \right),
    \quad \mathbf{Q} \succ \mathbf{0}, \text{i.e.}, \mathbf{Q} \in \mathbf{S}_d^{++},
    \quad \mathbf{h} \in \mathbb{R}^d,
    \quad c \in \mathbb{R}
\end{align*}

\paragraph{Solution:}

The gradient of \( f \) can be computed as
\begin{equation*}
    \nabla f(\mathbf{x}) = \mathbf{Q} \mathbf{x} + \mathbf{h}
\end{equation*}
and thereby from the first-order necessary condition for local minimum, the local minimum \( \mathbf{x}^* \) must necessarily satisfy
\begin{equation*}
    \nabla f(\mathbf{x}^*) = \mathbf{Q} \mathbf{x}^* + \mathbf{h} = \mathbf{0}
    \quad \iff \quad
    \boxed{
        \mathbf{x}^* = -\mathbf{Q}^{-1} \mathbf{h}
    }
\end{equation*}
