\begin{tabular}{ll}

    \( \R^{d \times d} \)
    &
    Set of all real \( d \times d \) matrices
    \\

    \( \SD \)
    &
    Set of all real \( d \times d \) symmetric matrices,
    \( \SD = \{ \mathbf{A} \in \R^{d \times d} \mid \mathbf{A} = \mathbf{A}^\top \} \)
    \\

    \( \PSD \)
    &
    Set of all real \( d \times d \) symmetric positive semi-definite matrices,
    \( \PSD = \{ \mathbf{A} \in \SD \mid \mathbf{A} \succeq \zero \} \)
    \\

    \( \PD \)
    &
    Set of all real \( d \times d \) symmetric positive definite matrices,
    \( \PD = \{ \mathbf{A} \in \SD \mid \mathbf{A} \succ \zero \} \)

\end{tabular}

\section{Little-o notation}

(Landau's notation~\cite{apostol1991calculus})
A function \( f: \R^d \to \R \) is said to be in \( o(g(\mathbf{x})) \) as \( \mathbf{x} \to \mathbf{x}_0 \), for some function \( g: \R^d \to \R \), \( g(\mathbf{x}) \neq 0 \) in a neighborhood of \( \mathbf{x}_0 \), if
\begin{equation*}
    \lim_{\mathbf{x} \to \mathbf{x}_0} \frac{f(\mathbf{x})}{g(\mathbf{x})} = 0
    \quad \iff \quad
    \forall \epsilon > 0,
    \exists \delta > 0,
    \text{ s.t. }
    \abs{\frac{f(\mathbf{x})}{g(\mathbf{x})}} < \epsilon,
    \quad \forall \mathbf{x} \in B_{\delta}(\mathbf{x}_0) \setminus \{ \mathbf{x}_0 \}
\end{equation*}

\subsection{Properties}

\begin{itemize}
    \item \( f \in o(g) \implies c \cdot f \in o(g), \; \forall c \in \R \).

    \item \( f \in o(F), \;  g \in o(G) \implies f \cdot g \in o(F \cdot G) \).

    \item \( f \in o(h), \; g \in o(h) \implies f + g \in o(h) \).

    \item \( f \in o(g), \; g \in o(h) \implies f \in o(h) \).

    \item \( f(\mathbf{x}) \in o(1) \text{ as } \mathbf{x} \to \mathbf{x}_0 \implies f(\mathbf{x}) \to 0 \text{ as } \mathbf{x} \to \mathbf{x}_0 \).

    \item \( |f(\mathbf{x})| \leq |h(\mathbf{x})| \text{ near } \mathbf{x}_0, \; h \in o(g) \implies f \in o(g) \).

    \item \( f \in o(g), \; \phi: \R \to \R \text{ is continuous at } 0 \text{ with } \phi(0) = 0 \implies \phi(f(\mathbf{x})) \in o(\phi(g(\mathbf{x}))) \).

    \item \( \alpha > \beta > 0 \implies \|\mathbf{x} - \mathbf{x}_0\|^\alpha \in o(\|\mathbf{x} - \mathbf{x}_0\|^\beta) \text{ as } \mathbf{x} \to \mathbf{x}_0 \).
\end{itemize}
