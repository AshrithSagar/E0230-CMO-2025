\section{Little-o notation}

(Landau's notation~\cite{apostol1991calculus})
A function \( f: \R^d \to \R \) is said to be in \( \smalloh{g(\x)} \) as \( \x \to \x_0 \), for some function \( g: \R^d \to \R \), \( g(\x) \neq 0 \) in a neighborhood of \( \x_0 \), if
\begin{equation*}
    f \in \smalloh{g},
    \; \x \to \x_0
    \;\; \iff \;\;
    \lim_{\x \to \x_0} \frac{f(\x)}{g(\x)} = 0
    \;\; \iff \;\;
    \forall \epsilon > 0,
    \exists \delta > 0,
    \text{ s.t. }
    \abs{\frac{f(\x)}{g(\x)}} < \epsilon,
    \; \forall \x \in B_{\delta}(\x_0) \setminus \set{\x_0}
\end{equation*}

\subsection{Properties}

\begin{itemize}
    \item \( f \in \smalloh{g} \implies c \cdot f \in \smalloh{g}, \; \forall c \in \R \).

    \item \( f \in \smalloh{F}, \;  g \in \smalloh{G} \implies f \cdot g \in \smalloh{F \cdot G} \).

    \item \( f \in \smalloh{h}, \; g \in \smalloh{h} \implies f + g \in \smalloh{h} \).

    \item \( f \in \smalloh{g}, \; g \in \smalloh{h} \implies f \in \smalloh{h} \).

    \item \( f(\x) \in \smalloh{1} \text{ as } \x \to \x_0 \implies f(\x) \to 0 \text{ as } \x \to \x_0 \).

    \item \( |f(\x)| \leq |h(\x)| \text{ near } \x_0, \; h \in \smalloh{g} \implies f \in \smalloh{g} \).

    \item \( f \in \smalloh{g}, \; \phi: \R \to \R \text{ is continuous at } 0 \text{ with } \phi(0) = 0 \implies \phi(f(\x)) \in \smalloh{\phi(g(\x))} \).

    \item \( \alpha > \beta > 0 \implies \norm{\x - \x_0}^\alpha \in \smalloh{\norm{\x - \x_0}^\beta} \text{ as } \x \to \x_0 \).
\end{itemize}

\section{Multi-index notation}\label{sec:multi-index-notation}

A \( d \)--dimensional \textbf{""multi-index""} \( \bfalpha \) is an ordered \( d \)-tuple of non-negative integers, i.e.,
\begin{equation*}
    \bfalpha
    \triangleq
    (\alpha_1, \alpha_2, \ldots, \alpha_d),
    \; \alpha_i \in \N,
    \; i \in \set{1, 2, \ldots, d}
\end{equation*}

\begin{align*}
    \abs{\bfalpha}
    & \triangleq
    \sum_{i = 1}^{d} \alpha_i
    =
    \alpha_1 + \alpha_2 + \cdots + \alpha_d
    \\
    \bfalpha!
    & \triangleq
    \prod_{i = 1}^{d} \alpha_i!
    =
    \alpha_1! \cdot \alpha_2! \cdots \alpha_d!
    \\
    {\x}^{\bfalpha}
    & \triangleq
    \prod_{i = 1}^{d} x_i^{\alpha_i}
    =
    x_1^{\alpha_1} \cdot x_2^{\alpha_2} \cdots x_d^{\alpha_d},
    \quad \x \in \R^d
    \\
    \partial^{\bfalpha}
    & \triangleq
    \frac
    {\partial^{\abs{\bfalpha}}}
    {\partial x_1^{\alpha_1} \partial x_2^{\alpha_2} \cdots \partial x_d^{\alpha_d}}
    =
    \frac
    {\partial^{\alpha_1 + \alpha_2 + \cdots + \alpha_d}}
    {\partial x_1^{\alpha_1} \partial x_2^{\alpha_2} \cdots \partial x_d^{\alpha_d}}
\end{align*}
