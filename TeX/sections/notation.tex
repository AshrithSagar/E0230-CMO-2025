\section{Notation}

\begin{tabular}{ll}

    \( \mathbb{R}^{d \times d} \)
    &
    Set of all real \( d \times d \) matrices
    \\

    \( \mathbf{S}_{d} \)
    &
    Set of all real \( d \times d \) symmetric matrices,
    \( \mathbf{S}_{d} = \{ \mathbf{A} \in \mathbb{R}^{d \times d} \mid \mathbf{A} = \mathbf{A}^\top \} \)
    \\

    \( \mathbf{S}_{d}^{+} \)
    &
    Set of all real \( d \times d \) symmetric positive semi-definite matrices,
    \( \mathbf{S}_{d}^{+} = \{ \mathbf{A} \in \mathbf{S}_{d} \mid \mathbf{A} \succeq \mathbf{0} \} \)
    \\

    \( \mathbf{S}_{d}^{++} \)
    &
    Set of all real \( d \times d \) symmetric positive definite matrices,
    \( \mathbf{S}_{d}^{++} = \{ \mathbf{A} \in \mathbf{S}_{d} \mid \mathbf{A} \succ \mathbf{0} \} \)

\end{tabular}
