\begin{tabular}{ll}

    \( \R_{> 0} \)
    &
    Set of all positive real numbers, \( \R_{>0} = \set{x \in \R \given x > 0} \)
    \\

    \( \R_{\geq 0} \)
    &
    Set of all non-negative real numbers, \( \R_{\geq 0} = \set{x \in \R \given x \geq 0} \)
    \\

    \( \R^{d \times d} \)
    &
    Set of all real \( d \times d \) matrices, \( \R^{d \times d} = \set{{(a_{ij})}_{i, j = 1}^d \given a_{ij} \in \R, \; i,j = 1,\ldots,d} \)
    \\

    \( \SD \)
    &
    Set of all real \( d \times d \) symmetric matrices,
    \( \SD = \set{\A \in \R^{d \times d} \given \A = \A^\top} \)
    \\

    \( \PSD \)
    &
    Set of all real \( d \times d \) symmetric positive semi-definite matrices,
    \( \PSD = \set{\A \in \SD \given \A \succeq \zero} \)
    \\

    \( \PD \)
    &
    Set of all real \( d \times d \) symmetric positive definite matrices,
    \( \PD = \set{\A \in \SD \given \A \succ \zero} \)

\end{tabular}

\section{Little-o notation}

(Landau's notation~\cite{apostol1991calculus})
A function \( f: \R^d \to \R \) is said to be in \( o(g(\x)) \) as \( \x \to \x_0 \), for some function \( g: \R^d \to \R \), \( g(\x) \neq 0 \) in a neighborhood of \( \x_0 \), if
\begin{equation*}
    \lim_{\x \to \x_0} \frac{f(\x)}{g(\x)} = 0
    \quad \iff \quad
    \forall \epsilon > 0,
    \exists \delta > 0,
    \text{ s.t. }
    \abs{\frac{f(\x)}{g(\x)}} < \epsilon,
    \quad \forall \x \in B_{\delta}(\x_0) \setminus \set{\x_0}
\end{equation*}

\subsection{Properties}

\begin{itemize}
    \item \( f \in o(g) \implies c \cdot f \in o(g), \; \forall c \in \R \).

    \item \( f \in o(F), \;  g \in o(G) \implies f \cdot g \in o(F \cdot G) \).

    \item \( f \in o(h), \; g \in o(h) \implies f + g \in o(h) \).

    \item \( f \in o(g), \; g \in o(h) \implies f \in o(h) \).

    \item \( f(\x) \in o(1) \text{ as } \x \to \x_0 \implies f(\x) \to 0 \text{ as } \x \to \x_0 \).

    \item \( |f(\x)| \leq |h(\x)| \text{ near } \x_0, \; h \in o(g) \implies f \in o(g) \).

    \item \( f \in o(g), \; \phi: \R \to \R \text{ is continuous at } 0 \text{ with } \phi(0) = 0 \implies \phi(f(\x)) \in o(\phi(g(\x))) \).

    \item \( \alpha > \beta > 0 \implies \norm{\x - \x_0}^\alpha \in \smalloh{\norm{\x - \x_0}^\beta} \text{ as } \x \to \x_0 \).
\end{itemize}
