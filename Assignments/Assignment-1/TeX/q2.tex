\paragraph{Problem:}
We're given the following unconstrained optimisation problem: \( \min_{\mathbf{x} \in \mathbb{R}^5} f(\mathbf{x}) \), where we are given access to oracles that yield $f(\mathbf{x})$ and $\nabla f(\mathbf{x})$ for any $\mathbf{x} \in \mathbb{R}^5$.

\paragraph{(1). Gradient descent termination criterion:}

A typical termination criterion for gradient descent is based on the norm of the gradient, i.e., we terminate when
\begin{equation*}
    \|\nabla f(\mathbf{x}_k)\| \leq \epsilon,
    \quad
    \text{ for some small tolerance }
    \epsilon > 0
\end{equation*}
This indicates that we've (approximately) reached a stationary point, where the gradient is close to zero, suggesting a local minimum, maximum, or saddle point.

\subsubsection*{Sufficient decrease condition (Armijo condition)}

\begin{equation*}
    f \left( \mathbf{x}^{(k)} + \alpha_k \mathbf{p}^{(k)} \right)
    \leq
    f(\mathbf{x}^{(k)}) + c \alpha_k {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)},
    \quad 0 < c < 1
\end{equation*}

\subsubsection*{Goldstein conditions / Armijo-Goldstein conditions}

\vspace*{-1em}
\begin{align*}
    f(\mathbf{x}^{(k)}) + (1 - c) \alpha_k {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)}
    & \leq
    \overbrace{
        f \left( \mathbf{x}^{(k)} + \alpha_k \mathbf{p}^{(k)} \right)
        \leq
        f(\mathbf{x}^{(k)}) + c \alpha_k {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)}
    }^{\text{sufficient decrease}}
    \\
    \text{where } \;&
    0 < c < \frac{1}{2}
\end{align*}

\subsubsection*{Wolfe conditions}

The step length \( \alpha_k \) satisfies both the \textit{sufficient decrease condition} and the \textit{curvature condition}.
\begin{align*}
    f \left( \mathbf{x}^{(k)} + \alpha_k \mathbf{p}^{(k)} \right)
    & \leq
    f(\mathbf{x}^{(k)}) + c_1 \alpha_k {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)}
    \tag{sufficient decrease}
    \\
    {\nabla f \left( \mathbf{x}^{(k)} + \alpha_k \mathbf{p}^{(k)} \right)}^{\top} \mathbf{p}^{(k)}
    & \geq
    c_2 {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)}
    \tag{curvature}
    \\
    \text{where } \;&
    0 < c_1 < c_2 < 1
\end{align*}

\subsubsection*{Strong Wolfe conditions}

The step length \( \alpha_k \) satisfies both the \textit{sufficient decrease condition} and the \textit{strong curvature condition}.
\begin{align*}
    f \left( \mathbf{x}^{(k)} + \alpha_k \mathbf{p}^{(k)} \right)
    & \leq
    f(\mathbf{x}^{(k)}) + c_1 \alpha_k {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)}
    \tag{sufficient decrease}
    \\
    \left| {\nabla f \left( \mathbf{x}^{(k)} + \alpha_k \mathbf{p}^{(k)} \right)}^{\top} \mathbf{p}^{(k)} \right|
    & \leq
    c_2 \left| {\nabla f(\mathbf{x}^{(k)})}^{\top} \mathbf{p}^{(k)} \right|
    \tag{strong curvature}
    \\
    \text{where } \;&
    0 < c_1 < c_2 < 1
\end{align*}
