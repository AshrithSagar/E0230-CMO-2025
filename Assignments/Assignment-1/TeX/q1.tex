\section*{Question 1}

\paragraph{Problem:}
We're given the following unconstrained optimisation problem:
\begin{equation*}
    \min_{\mathbf{x} \in \mathbb{R}^2} f(\mathbf{x}),
    \quad
    f(\mathbf{x}) = \frac{1}{2} \mathbf{x}^T \mathbf{Q} \mathbf{x} + \mathbf{b}^T \mathbf{x},
    \quad
    \mathbf{Q} \succ 0,
    \quad
    b =
    \begin{bmatrix}
        1 \\ 1
    \end{bmatrix}
\end{equation*}
